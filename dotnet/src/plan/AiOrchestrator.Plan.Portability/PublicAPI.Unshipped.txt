#nullable enable
namespace AiOrchestrator.Plan.Portability;
public sealed class PlanExporter
public PlanExporter(IPlanStore store, IFileSystem fs, IClock clock, IOptionsMonitor<PortabilityOptions> opts);
public ValueTask<AbsolutePath> ExportAsync(PlanId planId, AbsolutePath outputFile, ExportOptions options, CancellationToken ct);
public sealed class PlanImporter
public PlanImporter(IPlanStore store, IFileSystem fs, IClock clock, IOptionsMonitor<PortabilityOptions> opts);
public ValueTask<PlanId> ImportAsync(AbsolutePath inputFile, ImportOptions options, CancellationToken ct);
public sealed record PortabilityOptions { public Version SchemaVersion { get; init; } = new Version(1, 0); }
public sealed record ExportOptions
public bool IncludeAttempts { get; init; } = false;
public bool IncludeArtifacts { get; init; } = false;   // auxiliary files referenced by jobs (e.g. instructions files)
public bool RedactPaths { get; init; } = true;   // strip absolute prefixes
public sealed record ImportOptions
public string? OverridePlanName { get; init; }
public string? OverrideTargetBranch { get; init; }
public ImportConflictPolicy IfPlanIdExists { get; init; } = ImportConflictPolicy.GenerateNewId;
public enum ImportConflictPolicy { Reject, GenerateNewId, OverwriteIfArchived }
public sealed class PortabilityArchive
public required Version SchemaVersion { get; init; }
public required DateTimeOffset CreatedAt { get; init; }
public required string AioVersion { get; init; }
public required Plan Plan { get; init; }   // serialized plan
public required ImmutableDictionary<string, byte[]> Artifacts { get; init; }   // file_name -> content
public sealed class PortabilitySchemaMismatchException : Exception { public required Version Expected { get; init; } public required Version Actual { get; init; } }
public sealed class ImportConflictException : Exception { public required PlanId ExistingPlanId { get; init; } public required PlanStatus ExistingStatus { get; init; } }
