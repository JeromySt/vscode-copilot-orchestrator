#nullable enable
namespace AiOrchestrator.SkewManifest;
public sealed class SkewManifestObserver : IHostedService, IAsyncDisposable
public SkewManifestObserver(IHttpClientFactory http, IFileSystem fs, IClock clock, IAuditLog audit, IEventBus bus, IOptionsMonitor<SkewManifestOptions> opts, ILogger<SkewManifestObserver> logger);
public Task StartAsync(CancellationToken ct);
public Task StopAsync(CancellationToken ct);
public ValueTask<SkewManifest?> CurrentAsync(CancellationToken ct);
public ValueTask<SkewManifestVerification> VerifyAsync(SkewManifest manifest, CancellationToken ct);
public sealed record SkewManifestOptions
public string ManifestUrl { get; init; } = "https://aka.ms/aio-build-keys.json";
public TimeSpan PollInterval { get; init; } = TimeSpan.FromHours(6);
public TimeSpan StaleAfter { get; init; } = TimeSpan.FromDays(30);
public int RequiredHsmSignatures { get; init; } = 3;   // M-of-N where N=5
public ImmutableArray<byte[]> KnownHsmPublicKeys { get; init; }   // 5 burn-in HSM pubkeys
public ImmutableArray<byte[]> EmergencyRevocationPublicKeys { get; init; }
public string? TransparencyLogUrl { get; init; }   // Sigstore-style
public sealed record SkewManifest
public required Version SchemaVersion { get; init; }   // "1"
public required Version ManifestVersion { get; init; }
public required DateTimeOffset SignedAt { get; init; }
public required DateTimeOffset NotValidAfter { get; init; }
public required ImmutableArray<TrustedAuditPubKey> TrustedAuditPubKeys { get; init; }
public required ImmutableArray<HsmSignature> HsmSignatures { get; init; }   // ≥ M of N
public required EmergencyRevocation? EmergencyRevocation { get; init; }
public required string? TransparencyLogProof { get; init; }
public sealed record TrustedAuditPubKey { public required string KeyId { get; init; } public required byte[] PublicKey { get; init; } public required DateTimeOffset NotValidBefore { get; init; } public required DateTimeOffset NotValidAfter { get; init; } public required string? RevocationReason { get; init; } }
public sealed record HsmSignature { public required string HsmId { get; init; } public required byte[] Signature { get; init; } public required string Algorithm { get; init; } /* "Ed25519" */ }
public sealed record EmergencyRevocation { public required ImmutableArray<string> RevokedKeyIds { get; init; } public required string Reason { get; init; } public required DateTimeOffset RevokedAt { get; init; } public required ImmutableArray<HsmSignature> AdditionalSignatures { get; init; }   /* signed by EmergencyRevocationPublicKeys */ }
public sealed record SkewManifestVerification { public required bool Ok { get; init; } public required SkewManifestRejectionReason? Reason { get; init; } public required string? Detail { get; init; } }
public enum SkewManifestRejectionReason { InsufficientSignatures, InvalidSignature, ExpiredManifest, VersionRegression, TransparencyLogMismatch, EmergencyRevocationInvalid, UnknownHsmSigner }
internal sealed class HsmSignatureVerifier
public HsmSignatureVerifier(IOptionsMonitor<SkewManifestOptions> opts);
public bool TryVerify(SkewManifest mfst, out int validCount, out string? failureDetail);   // verifies that >=M signatures from known HSMs are valid
internal sealed class TransparencyLogChecker
public TransparencyLogChecker(IHttpClientFactory http, IClock clock, IOptionsMonitor<SkewManifestOptions> opts);
public ValueTask<TransparencyLogCheckResult> CheckAsync(SkewManifest mfst, CancellationToken ct);   // submits inclusion proof check to log
public sealed record TransparencyLogCheckResult { public required bool Included { get; init; } public required string? FailureReason { get; init; } }
