#nullable enable
namespace AiOrchestrator.Output;
public sealed class StreamRedirector : IAsyncDisposable
public StreamRedirector(IEventStore log, IEventBus bus, LineProjector lineView, IClock clock, IOptionsMonitor<RedirectorOptions> opts);
public ValueTask AttachAsync(JobId job, RunId run, PipeReader stdout, PipeReader stderr, CancellationToken ct);
public ValueTask DetachAsync(JobId job, CancellationToken ct);
public OutputSnapshot SnapshotFor(JobId job);
public sealed record RedirectorOptions { [Range(1024, 1024 * 1024)] public int RingBufferBytes { get; init; } = 256 * 1024; public int PerConsumerQueueDepth { get; init; } = 1024; }
public sealed class OutputSnapshot { public required ImmutableArray<byte> RecentStdoutBytes { get; init; } public required ImmutableArray<byte> RecentStderrBytes { get; init; } public required long TotalStdoutBytes { get; init; } public required long TotalStderrBytes { get; init; } }
internal sealed class FanoutBroker
public FanoutBroker(IOptionsMonitor<RedirectorOptions> opts, IEventBus bus);
public IAsyncDisposable Subscribe(JobId job, OutputConsumerKind kind, Func<OutputChunk, CancellationToken, ValueTask> handler);
public ValueTask PublishAsync(JobId job, OutputChunk chunk, CancellationToken ct);
public enum OutputConsumerKind { EventLog, LineView, RingBuffer, Logger }
public sealed record OutputChunk { public required JobId JobId { get; init; } public required OutputStream Stream { get; init; } public required long ByteOffset { get; init; } public required ReadOnlyMemory<byte> Data { get; init; } public required DateTimeOffset At { get; init; } }
public enum OutputStream { StdOut, StdErr }
public sealed class ConsumerLagged
public required JobId JobId { get; init; }
public required OutputConsumerKind Consumer { get; init; }
public required long DroppedBytes { get; init; }
public required DateTimeOffset At { get; init; }
