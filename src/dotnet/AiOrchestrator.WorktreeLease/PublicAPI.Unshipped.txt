#nullable enable

namespace AiOrchestrator.WorktreeLease;
public sealed class WorktreeLeaseManager : IWorktreeLease
public WorktreeLeaseManager(IFileSystem fs, IClock clock, IEventBus bus, IOptionsMonitor<LeaseOptions> opts, ILogger<WorktreeLeaseManager> logger);
public ValueTask<LeaseHandle> AcquireAsync(AbsolutePath worktree, AuthContext holder, TimeSpan ttl, CancellationToken ct);
public ValueTask RenewAsync(LeaseHandle handle, TimeSpan ttl, CancellationToken ct);
public ValueTask ReleaseAsync(LeaseHandle handle, CancellationToken ct);
public ValueTask<LeaseInfo?> InspectAsync(AbsolutePath worktree, CancellationToken ct);
public sealed record LeaseOptions { public TimeSpan StaleCheckInterval { get; init; } = TimeSpan.FromSeconds(10); public TimeSpan AcquireTimeout { get; init; } = TimeSpan.FromSeconds(30); public TimeSpan AcquireRetryDelay { get; init; } = TimeSpan.FromMilliseconds(250); }
public sealed class LeaseHandle : IAsyncDisposable
public required FencingToken Token { get; init; }
public required AbsolutePath Worktree { get; init; }
public required AuthContext Holder { get; init; }
public required DateTimeOffset ExpiresAt { get; init; }
public ValueTask DisposeAsync();   // calls ReleaseAsync
public readonly record struct FencingToken(long Value) : IComparable<FencingToken>
public int CompareTo(FencingToken other) => Value.CompareTo(other.Value);
public override string ToString() => Value.ToString(System.Globalization.CultureInfo.InvariantCulture);
public sealed record LeaseInfo { public required FencingToken Token { get; init; } public required AuthContext Holder { get; init; } public required DateTimeOffset ExpiresAt { get; init; } public required DateTimeOffset AcquiredAt { get; init; } }
internal sealed class CasLeaseStore
public CasLeaseStore(IFileSystem fs, IClock clock);
public ValueTask<bool> TryWriteAsync(AbsolutePath leaseFile, LeaseFileContent content, FencingToken expectedPriorToken, CancellationToken ct);   // Opens with FileShare.None; reads existing token; if expectedPriorToken != stored, returns false
public ValueTask<LeaseFileContent?> ReadAsync(AbsolutePath leaseFile, CancellationToken ct);
public ValueTask<bool> EnforceWriteWithTokenAsync(AbsolutePath worktree, FencingToken token, RepoRelativePath relativeFile, ReadOnlyMemory<byte> contents, CancellationToken ct);   // LS-INF-1: rejects if stored token != provided token
public sealed record LeaseFileContent { public required FencingToken Token { get; init; } public required string HolderUserName { get; init; } public required string HolderProcessHash { get; init; } public required DateTimeOffset AcquiredAt { get; init; } public required DateTimeOffset ExpiresAt { get; init; } public required string SchemaVersion { get; init; } /* "1" */ }
public sealed class StaleLeaseDetector : IAsyncDisposable
public StaleLeaseDetector(IFileSystem fs, IClock clock, IEventBus bus, IOptionsMonitor<LeaseOptions> opts);
public ValueTask StartAsync(AbsolutePath worktree, FencingToken expectedToken, CancellationToken ct);
public sealed class WorktreeLeaseStolen { public required AbsolutePath Worktree { get; init; } public required FencingToken ExpectedToken { get; init; } public required FencingToken ObservedToken { get; init; } public required DateTimeOffset At { get; init; } }
public sealed class StaleLeaseTokenException : Exception { public required FencingToken ProvidedToken { get; init; } public required FencingToken StoredToken { get; init; } }
