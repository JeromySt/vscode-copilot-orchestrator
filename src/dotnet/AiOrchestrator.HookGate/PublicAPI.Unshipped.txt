#nullable enable
namespace AiOrchestrator.HookGate;
public sealed class HookGateDaemon : IHostedService, IAsyncDisposable
public HookGateDaemon(INonceManager nonces, IRpcServer rpc, IRedirectionManager redirect, IClock clock, IEventBus bus, IOptionsMonitor<HookGateOptions> opts, ILogger<HookGateDaemon> logger);
public Task StartAsync(CancellationToken ct);
public Task StopAsync(CancellationToken ct);
public sealed record HookGateOptions
public TimeSpan NonceRotation { get; init; } = TimeSpan.FromMinutes(5);
public AbsolutePath SocketPath { get; init; } = AbsolutePath.Of("/run/ai-orchestrator/hookgate.sock");
public string PipeName { get; init; } = @"\\.\pipe\AiOrchestratorHookGate";
public bool RequireImmutability { get; init; } = false;   // if true, emits warning when BEST-EFFORT immutability is unsupported
public TimeSpan ApprovalTokenTtl { get; init; } = TimeSpan.FromMinutes(2);
public interface IHookGateClient
ValueTask<HookApproval> CheckInAsync(HookCheckInRequest request, CancellationToken ct);
public sealed record HookCheckInRequest { public required HookKind Kind { get; init; } public required RepoRelativePath HookFile { get; init; } public required AbsolutePath WorktreeRoot { get; init; } public required AuthContext Principal { get; init; } public required ImmutableDictionary<string, string> Env { get; init; } }
public enum HookKind { PreCommit, CommitMsg, PrePush, PostCheckout, PreRebase, PostMerge, PreReceive, Update, PostReceive, PostUpdate, PrepareCommitMsg }
public sealed record HookApproval { public required string TokenId { get; init; } public required byte[] Hmac { get; init; } public required DateTimeOffset IssuedAt { get; init; } public required DateTimeOffset ExpiresAt { get; init; } public required AuthContext Principal { get; init; } }
public interface INonceManager
Nonce Current { get; }
event EventHandler<NonceRotated> Rotated;
public sealed record Nonce { public required string Value { get; init; } public required DateTimeOffset IssuedAt { get; init; } public required DateTimeOffset RotatesAt { get; init; } }
public interface IRedirectionManager
ValueTask InstallRedirectionAsync(AbsolutePath gitHooksDir, AbsolutePath canonicalDispatcherPath, CancellationToken ct);
ValueTask UninstallRedirectionAsync(AbsolutePath gitHooksDir, CancellationToken ct);
ValueTask<RedirectionMode> GetActiveModeAsync(AbsolutePath gitHooksDir, CancellationToken ct);
public enum RedirectionMode { BindMount, Junction, Symlink, NotInstalled }
internal sealed class LinuxRedirectionManager : IRedirectionManager
internal sealed class WindowsRedirectionManager : IRedirectionManager
internal sealed class ImmutabilityProbe
public ValueTask<ImmutabilityResult> ProbeAsync(AbsolutePath path, CancellationToken ct);   // Linux: try chattr +i; macOS: chflags uchg; Windows: DACL-deny WRITE
public bool IsImmutabilitySupported(ImmutabilityResult result);
public HookGateNonceImmutabilityUnsupported BuildEvent(AbsolutePath path, ImmutabilityResult result);
public sealed record ImmutabilityResult { public required bool Supported { get; init; } public required string Mechanism { get; init; } public required string? FailureReason { get; init; } }
public sealed class HookGateNonceImmutabilityUnsupported { public required AbsolutePath Path { get; init; } public required string Mechanism { get; init; } public required string Reason { get; init; } public required DateTimeOffset At { get; init; } }
internal sealed class LinkValidator
public ValueTask<LinkValidationResult> ValidateAsync(AbsolutePath hookFile, CancellationToken ct);   // Linux: stat → check st_nlink == 1, check resolved path inside worktree; Windows: GetFileInformationByHandle → reparse-point check
public sealed record LinkValidationResult { public required bool Ok { get; init; } public required string? FailureReason { get; init; } }
public sealed class HookApprovalDeniedException : Exception { public required string Reason { get; init; } public required HookKind Kind { get; init; } }
