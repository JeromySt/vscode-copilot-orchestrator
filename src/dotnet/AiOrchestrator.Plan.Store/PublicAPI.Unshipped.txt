#nullable enable
namespace AiOrchestrator.Plan.Store;
public sealed class PlanStore : IPlanStore, IAsyncDisposable
public PlanStore(AbsolutePath storeRoot, IFileSystem fs, IClock clock, IEventBus bus, IOptionsMonitor<PlanStoreOptions> opts, ILogger<PlanStore> logger);
public ValueTask<PlanId> CreateAsync(Plan initialPlan, IdempotencyKey idemKey, CancellationToken ct);
public ValueTask<Plan?> LoadAsync(PlanId id, CancellationToken ct);
public ValueTask MutateAsync(PlanId id, PlanMutation mutation, IdempotencyKey idemKey, CancellationToken ct);
public ValueTask CheckpointAsync(PlanId id, CancellationToken ct);
public IAsyncEnumerable<Plan> ListAsync(CancellationToken ct);
public IAsyncEnumerable<PlanMutation> ReadJournalAsync(PlanId id, long fromSeq, CancellationToken ct);
public IAsyncEnumerable<Plan> WatchAsync(PlanId id, CancellationToken ct);   // SUB-3: replay snapshot then live mutations
public sealed record PlanStoreOptions { public int CheckpointAfterMutations { get; init; } = 100; public TimeSpan CheckpointAfterTime { get; init; } = TimeSpan.FromMinutes(5); }
public abstract record PlanMutation { public required long Seq { get; init; } public required IdempotencyKey IdemKey { get; init; } public required DateTimeOffset At { get; init; } }
public sealed record JobAdded(long Seq, IdempotencyKey IdemKey, DateTimeOffset At, JobNode Node) : PlanMutation { }
public sealed record JobRemoved(long Seq, IdempotencyKey IdemKey, DateTimeOffset At, JobId JobId) : PlanMutation { }
public sealed record JobDepsUpdated(long Seq, IdempotencyKey IdemKey, DateTimeOffset At, JobId JobId, ImmutableArray<JobId> NewDeps) : PlanMutation { }
public sealed record JobStatusUpdated(long Seq, IdempotencyKey IdemKey, DateTimeOffset At, JobId JobId, JobStatus NewStatus, JobPhase NewPhase) : PlanMutation { }
public sealed record JobAttemptRecorded(long Seq, IdempotencyKey IdemKey, DateTimeOffset At, JobId JobId, JobAttempt Attempt) : PlanMutation { }
public sealed record PlanStatusUpdated(long Seq, IdempotencyKey IdemKey, DateTimeOffset At, PlanStatus NewStatus) : PlanMutation { }
public readonly record struct IdempotencyKey(string Value)
public static IdempotencyKey FromContent(ReadOnlySpan<byte> content);   // SHA-256 of canonical JSON
public static IdempotencyKey FromGuid(Guid value);
internal sealed class PlanJournal : IAsyncDisposable
public PlanJournal(AbsolutePath path, IFileSystem fs, IClock clock);
public ValueTask AppendAsync(PlanMutation mutation, CancellationToken ct);
public IAsyncEnumerable<PlanMutation> ReadFromAsync(long fromSeq, CancellationToken ct);
public ValueTask<bool> ContainsIdempotencyKeyAsync(IdempotencyKey key, CancellationToken ct);   // RW-2-IDEM lookup
internal sealed class PlanCheckpointer
public PlanCheckpointer(AbsolutePath path, IFileSystem fs);
public ValueTask WriteAsync(Plan plan, long upToSeq, CancellationToken ct);   // atomic temp+rename
public ValueTask<(Plan plan, long upToSeq)?> LoadLatestAsync(CancellationToken ct);
public sealed class IdempotencyConflictException : Exception { public required IdempotencyKey Key { get; init; } public required PlanMutation StoredMutation { get; init; } public required PlanMutation NewMutation { get; init; } }
