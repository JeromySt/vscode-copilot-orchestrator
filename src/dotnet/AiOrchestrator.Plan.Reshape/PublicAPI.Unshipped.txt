#nullable enable
namespace AiOrchestrator.Plan.Reshape;
public sealed class PlanReshaper
public PlanReshaper(IPlanStore store, IClock clock, IOptionsMonitor<PlanReshapeOptions> opts);
public ValueTask<ReshapeResult> ApplyAsync(PlanId planId, ImmutableArray<ReshapeOperation> ops, IdempotencyKey idemKey, CancellationToken ct);
public sealed record PlanReshapeOptions { public int MaxOpsPerCall { get; init; } = 100; }
public abstract record ReshapeOperation { }
public sealed record AddJob(JobNode Spec, ImmutableArray<JobId> Dependencies) : ReshapeOperation { }
public sealed record RemoveJob(JobId TargetJobId) : ReshapeOperation { }
public sealed record UpdateDeps(JobId TargetJobId, ImmutableArray<JobId> NewDependencies) : ReshapeOperation { }
public sealed record AddBefore(JobId ExistingJobId, JobNode NewJobSpec, ImmutableArray<JobId> NewJobDependencies) : ReshapeOperation { }
public sealed record AddAfter(JobId ExistingJobId, JobNode NewJobSpec) : ReshapeOperation { }
public sealed record ReshapeResult { public required ImmutableArray<OperationResult> PerOperation { get; init; } public required Plan UpdatedPlan { get; init; } }
public sealed record OperationResult { public required ReshapeOperation Op { get; init; } public required bool Success { get; init; } public required string? FailureReason { get; init; } public required JobId? AffectedJobId { get; init; } }
internal sealed class CycleGuard
public CycleResult WouldCreateCycle(PlanGraph current, ReshapeOperation op);
public sealed record CycleResult { public required bool Cycle { get; init; } public required ImmutableArray<JobId>? Cycle_ { get; init; } }
public sealed class ReshapeRejectedException : Exception { public required ImmutableArray<OperationResult> Failures { get; init; } }
public sealed class SnapshotValidationNodeImmutableException : Exception { }
