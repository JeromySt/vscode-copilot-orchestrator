namespace AiOrchestrator.Plan.Models;

public sealed record Plan
{
    public required PlanId Id { get; init; }
    public required string Name { get; init; }
    public required string BaseBranch { get; init; }
    public required string TargetBranch { get; init; }
    public required PlanStatus Status { get; init; }
    public required DateTimeOffset CreatedAt { get; init; }
    public required DateTimeOffset UpdatedAt { get; init; }
    public required PlanLimits Limits { get; init; }
    public required ImmutableArray<JobNode> Jobs { get; init; }
    public required ImmutableArray<JobEdge> Edges { get; init; }
    public required ImmutableDictionary<string, string> Env { get; init; }
    public required PlanVerification Verification { get; init; }
}
public enum PlanStatus { Scaffolding, PendingStart, Pending, Running, Paused, Succeeded, Partial, Failed, Canceled }

public sealed record PlanLimits { public required int MaxParallel { get; init; } public required int MaxJobs { get; init; } public required long MaxDiskBytes { get; init; } public required TimeSpan MaxPlanDuration { get; init; } }

public sealed record JobNode
{
    public required JobId Id { get; init; }
    public required string ProducerId { get; init; }
    public required string Name { get; init; }
    public required string Task { get; init; }
    public required WorkSpec Work { get; init; }
    public WorkSpec? Prechecks { get; init; }
    public WorkSpec? Postchecks { get; init; }
    public bool AutoHeal { get; init; } = true;
    public bool ExpectsNoChanges { get; init; } = false;
    public string? Group { get; init; }
    public required JobStatus Status { get; init; }
    public required JobPhase CurrentPhase { get; init; }
    public required ImmutableArray<JobAttempt> Attempts { get; init; }
    public required DateTimeOffset CreatedAt { get; init; }
    public CommitSha? AccumulatedCommitSha { get; init; }
}
public enum JobPhase { Setup, Prechecks, Work, Postchecks, Commit, ForwardIntegration, Done }

public sealed record JobEdge { public required JobId From { get; init; } public required JobId To { get; init; } }

public sealed record JobAttempt { public required int AttemptNumber { get; init; } public required RunId RunId { get; init; } public required DateTimeOffset StartedAt { get; init; } public required DateTimeOffset? EndedAt { get; init; } public required JobStatus FinalStatus { get; init; } public required JobPhase EndedAtPhase { get; init; } public required string? FailureReason { get; init; } public required CommitSha? CommitSha { get; init; } }

public sealed record PlanVerification { public WorkSpec? VerifyRi { get; init; } public bool SnapshotValidationAutoInjected { get; init; } = true; }

public sealed class PlanGraph
{
    public PlanGraph(Plan plan);
    public IReadOnlyDictionary<JobId, JobNode> NodesById { get; }
    public IReadOnlyDictionary<JobId, ImmutableArray<JobId>> Predecessors { get; }
    public IReadOnlyDictionary<JobId, ImmutableArray<JobId>> Successors { get; }
    public ImmutableArray<JobId> Roots { get; }
    public ImmutableArray<JobId> Leaves { get; }
    public bool IsAcyclic(out ImmutableArray<JobId> cycle);
    public ImmutableArray<JobId> TopologicalOrder();   // throws on cycle
}

### Behavioral invariants

1. **INV-1** — All records `sealed record`; no mutable setters.
2. **INV-2** — `JobStatus` transitions enforced via `JobStatus.AllowedTransitions` from job 001.
3. **INV-3** — `PlanGraph` constructor builds adjacency in O(V+E); rejects duplicate edges.
4. **INV-4** — `IsAcyclic(out cycle)` returns the offending cycle vertex sequence on failure.
5. **INV-5** — `TopologicalOrder()` is deterministic given the same input (ties broken by `JobId.Value`).
6. **INV-6** — `Plan` is fully serializable via the `ModelsSerializationContext` (job 001 + extension here).
7. **INV-7** — `PlanLimits` defaults: `MaxParallel = 8`, `MaxJobs = 128`, `MaxDiskBytes = 8 GiB`, `MaxPlanDuration = 24 h`.
8. **INV-8** — `Plan.UpdatedAt` is the latest of all `Jobs[].Attempts[].EndedAt` and the plan creation time.
