### Public API surface (exact)

```csharp
namespace AiOrchestrator.Models.Ids;

public readonly record struct PlanId(Guid Value)
{
    public static PlanId New() => new(Guid.NewGuid());
    public override string ToString() => $"plan_{Value:N}";
    public static PlanId Parse(string s);
    public static bool TryParse(string s, out PlanId id);
}
public readonly record struct JobId(Guid Value)  { /* same shape, prefix "job_" */ }
public readonly record struct RunId(Guid Value)  { /* prefix "run_" */ }
public readonly record struct CommitSha(string Hex) { /* validates 40-char hex */ }

namespace AiOrchestrator.Models.Paths;
public readonly record struct AbsolutePath
{
    public string Value { get; }
    public AbsolutePath(string value); // throws ArgumentException if not Path.IsPathRooted
    public AbsolutePath Combine(RelativePath rel);
    public override string ToString() => Value;
}
public readonly record struct RelativePath { /* refuses leading '/' or 'X:\\' */ }
public readonly record struct RepoRelativePath { /* refuses '..' segments */ }

namespace AiOrchestrator.Models;

public abstract record WorkSpec
{
    public required string Producer { get; init; }
    public required string Description { get; init; }
}
public sealed record AgentSpec : WorkSpec { public required AgentKind Kind { get; init; } public required string Prompt { get; init; } public string? Model { get; init; } public TimeSpan? Timeout { get; init; } }
public sealed record ShellSpec : WorkSpec { public required ShellKind Shell { get; init; } public required string Script { get; init; } public RepoRelativePath? WorkingDirectory { get; init; } }
public sealed record ProcessSpec : WorkSpec { public required string Executable { get; init; } public required ImmutableArray<string> Arguments { get; init; } public ImmutableDictionary<string, string>? Environment { get; init; } }

public enum AgentKind { ClaudeCode, CodexCli, GeminiCli, CopilotCli, GhCopilot, Qwen }
public enum ShellKind { Bash, Sh, Cmd, PowerShell, Pwsh }

public enum JobStatus { Pending, Ready, Scheduled, Running, Succeeded, Failed, Blocked, Canceled }
public enum Phase { Setup, Prechecks, Work, Postchecks, Commit, Teardown }

namespace AiOrchestrator.Models.Auth;
public sealed record AuthContext { public required string PrincipalId { get; init; } public required string DisplayName { get; init; } public required ImmutableArray<string> Scopes { get; init; } public DateTimeOffset IssuedAtUtc { get; init; } public DateTimeOffset? ExpiresAtUtc { get; init; } }

namespace AiOrchestrator.Models.Eventing;
public sealed record EventEnvelope { public required Guid EventId { get; init; } public required long RecordSeq { get; init; } public required DateTimeOffset OccurredAtUtc { get; init; } public required string EventType { get; init; } public required int SchemaVersion { get; init; } public required JsonElement Payload { get; init; } public PlanId? PlanId { get; init; } public JobId? JobId { get; init; } public string? PrincipalId { get; init; } }

namespace AiOrchestrator.Models.Redaction;
public sealed record RedactionPolicy { public required ImmutableArray<string> EnabledRules { get; init; } public required PseudonymizationMode PseudonymizationMode { get; init; } }
public enum PseudonymizationMode { Off, Anonymous, Reversible }

[JsonSerializable(typeof(WorkSpec))]
[JsonSerializable(typeof(AgentSpec))]
[JsonSerializable(typeof(ShellSpec))]
[JsonSerializable(typeof(ProcessSpec))]
[JsonSerializable(typeof(EventEnvelope))]
[JsonSerializable(typeof(AuthContext))]
[JsonSerializable(typeof(RedactionPolicy))]
public sealed partial class ModelsSerializationContext : JsonSerializerContext;
```

### Behavioral invariants

1. **INV-1** — Every Id type's `ToString()` is round-trippable via `Parse`.
2. **INV-2** — `AbsolutePath` constructor throws on relative input; `RelativePath` throws on rooted input; `RepoRelativePath` throws on `..` segments after canonicalization.
3. **INV-3** — `WorkSpec` discriminator round-trips via `ModelsSerializationContext` (polymorphic via `[JsonDerivedType]`).
4. **INV-4** — `EventEnvelope.RecordSeq` is `long`; deserializing a `RecordSeq` ≤ 0 throws.
5. **INV-5** — Every record is `sealed` (no inheritance besides the explicit `WorkSpec` discriminator hierarchy).
6. **INV-6** — Zero references to `Microsoft.Extensions.*` or any other AiOrchestrator project.
7. **INV-7** — `JobStatus` transition table is encoded as a `static readonly FrozenSet<(JobStatus, JobStatus)> AllowedTransitions` matching §3.1.4 row-for-row.
