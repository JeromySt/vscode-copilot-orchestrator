### Public API surface

```csharp
namespace AiOrchestrator.Daemon;

public sealed class DaemonProgram
{
    public static Task<int> Main(string[] args);
    public static IHost BuildHost(string[] args);
}

public sealed record DaemonOptions
{
    public AbsolutePath PidFile { get; init; } = AbsolutePath.Of("/run/ai-orchestrator/aio-daemon.pid");
    public AbsolutePath InstallRoot { get; init; }
    public AbsolutePath UpdateStagingRoot { get; init; }
    public string ReleaseManifestUrl { get; init; } = "https://aka.ms/aio/release-manifest.signed.json";
    public TimeSpan UpdateCheckInterval { get; init; } = TimeSpan.FromHours(6);
    public bool RequireOfflineRootSignature { get; init; } = true;
}

public sealed class UpdateController : IHostedService
{
    public UpdateController(IHttpClientFactory http, IFileSystem fs, IClock clock, IAuditLog audit, IEventBus bus, IOptionsMonitor<DaemonOptions> opts, ILogger<UpdateController> logger);
    public Task StartAsync(CancellationToken ct);
    public Task StopAsync(CancellationToken ct);
    public ValueTask<UpdateOutcome> CheckAndApplyAsync(CancellationToken ct);
}
public enum UpdateOutcome { NoUpdate, Applied, Rejected_BadSignature, Rejected_VersionRegression, Rejected_DowngradeBlocked, Failed_Network, Failed_Disk, RolledBack }

internal sealed class ReleaseManifestFetcher
{
    public ValueTask<SignedReleaseManifest> FetchAsync(string url, CancellationToken ct);
    public ValueTask<bool> VerifyAsync(SignedReleaseManifest mfst, byte[] offlineRootPubKey, CancellationToken ct);   // verifies M-of-N HSM signatures (default 3-of-5)
}

public sealed record SignedReleaseManifest
{
    public required Version Version { get; init; }
    public required ImmutableArray<DaemonArtifact> Artifacts { get; init; }
    public required DateTimeOffset SignedAt { get; init; }
    public required ImmutableArray<HsmSignature> Signatures { get; init; }
    public required Version MinSupportedVersion { get; init; }
    public required ImmutableArray<byte[]> TrustedAuditPubKeys { get; init; }
}
public sealed record DaemonArtifact { public required string Filename { get; init; } public required string Sha256 { get; init; } public required long Bytes { get; init; } public required Uri DownloadUrl { get; init; } }
public sealed record HsmSignature { public required string KeyId { get; init; } public required byte[] Signature { get; init; } }

internal sealed class StagedSwap
{
    public ValueTask SwapAsync(AbsolutePath installRoot, AbsolutePath stagingRoot, CancellationToken ct);   // atomic rename of bin/ + symlink swap (Linux); junction swap (Windows)
    public ValueTask RollbackAsync(AbsolutePath installRoot, AbsolutePath previousBackupRoot, CancellationToken ct);
}

internal sealed class HealthCheck
{
    public ValueTask<HealthResult> RunAsync(AbsolutePath daemonExe, CancellationToken ct);   // post-update smoke test: spawns aio-daemon --selfcheck --json with 60 s timeout
}
public sealed record HealthResult { public required bool Ok { get; init; } public required string? FailureReason { get; init; } }
```

### Behavioral invariants

1. **INV-1** (UPD-RB-1) — `UpdateController` polls `ReleaseManifestUrl` every `UpdateCheckInterval` (default 6 h).
2. **INV-2** (UPD-RB-2) — Daemon NEVER applies an update without verifying the manifest's M-of-N HSM signatures against the offline-root public key (RequireOfflineRootSignature=true).
3. **INV-3** (UPD-RB-3) — Update flow: download to staging → verify hashes → backup current install → `StagedSwap.SwapAsync` → `HealthCheck.RunAsync` (60s smoke test) → on failure `StagedSwap.RollbackAsync`.
4. **INV-4** (UPD-RB-4) — `Rejected_VersionRegression` if `manifest.Version <= current`; `Rejected_DowngradeBlocked` if `current.Version < manifest.MinSupportedVersion`.
5. **INV-5** (UPD-RB-5) — Every outcome is audited via `IAuditLog`: `DaemonUpdateApplied`, `DaemonUpdateRejected`, `DaemonUpdateRolledBack`.
6. **INV-6** (UPD-RB-6) — Daemon emits `BuildKeyRolloverObserved` (job 016) when manifest's `TrustedAuditPubKeys` differ from previously-observed set; daemon NEVER signs anything with build keys (TRUST-ROOT-4 v1.4).
7. **INV-7** (UPD-RB-7) — Pid file written atomically (temp+rename, fsync); daemon refuses to start if a valid pid file exists for a running process.
8. **INV-8** — Graceful shutdown: SIGTERM (Unix) / Ctrl+Break (Windows) drains in-flight work for `ShutdownDrainTimeout` (default 30 s) before forcing.
